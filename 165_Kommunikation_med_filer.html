    • System.IO.Directory
    • System.IO.Path

// Slet

    Async IO

    Både de statiske metoder på File og Directory, samt stream har async metoder. Med async/await er det rimelige simpel kode:

    // Setup
string fil1 = @"c:\temp\f1.txt";
string fil2 = @"c:\temp\f2.txt";
System.IO.File.WriteAllText(fil1, "123");
System.IO.File.WriteAllText(fil2, "456");

string indhold = HentIndholdAsync(fil1, fil2).Result;
Console.WriteLine(indhold);

async Task<string> HentIndholdAsync(string f1, string f2)
{
    Task<string> r1 = System.IO.File.ReadAllTextAsync(f1);
    Task<string> r2 = System.IO.File.ReadAllTextAsync(f2);
    await Task.WhenAll(r1, r2);
    return r1.Result + r2.Result;
}


/*
 ---------- Output: ----------

123456

*/

    // Setup
string fil1 = @"c:\temp\f1.txt";
string fil2 = @"c:\temp\f2.txt";
System.IO.File.WriteAllText(fil1, "123");
System.IO.File.WriteAllText(fil2, "456");

string indhold = HentIndholdAsync(fil1, fil2).Result;
Console.WriteLine(indhold);

async Task<string> HentIndholdAsync(string f1, string f2)
{
    Task<string> r1, r2;
    System.IO.StreamReader sr1 = null;
    System.IO.StreamReader sr2 = null;
    try
    {
        sr1 = System.IO.File.OpenText(f1);
        r1 = sr1.ReadToEndAsync();
        sr2 = System.IO.File.OpenText(f2);
        r2 = sr2.ReadToEndAsync();
        await Task.WhenAll(r1, r2);
        return r1.Result + r2.Result;
    }
    catch (Exception)
    {
        sr1?.Dispose();
        sr2?.Dispose();
        sr1 = null;
        sr2 = null;
        // log
        throw;
    }
    finally
    {
        sr1?.Dispose();
        sr2?.Dispose();
        sr1 = null;
        sr2 = null;
    }
}

/*
 ---------- Output: ----------

123456

*/

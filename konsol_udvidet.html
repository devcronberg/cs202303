    En udvidet konsol applikation

    Hvis du ønsker at skabe en udvidet konsol applikation med både log og appsettings kan du følge denne guide.

    Opret

    Brug VS (eller dotnet.exe) til at oprette en simple konsol applikation (sæt kryds i “Do not use top-level-statements”). Det vil ende med:

    namespace MyConsoleApp
{
    internal class Program
    {
        static void Main(string[] args) 
        {
            // Set culture
            System.Threading.Thread.CurrentThread.CurrentCulture =
                new System.Globalization.CultureInfo("en-US");  // da-DK
            Console.WriteLine("Hello, world!");
        }
    }
}

    hvis du ikke ønsker at arbejde med argumenter ved start af applikationen så fjern string[] args:

    namespace MyConsoleApp
{
    internal class Program
    {
        static void Main() 
        {
            // Set culture
            System.Threading.Thread.CurrentThread.CurrentCulture =
                new System.Globalization.CultureInfo("en-US");  // da-DK            
            Console.WriteLine("Hello, world!");
        }
    }
}

    Hvis du ønsker en async app kan du rette void Main til async Task Main:

    namespace MyConsoleApp
{
    internal class Program
    {
        static async Task Main() 
        {
            // Set culture
            System.Threading.Thread.CurrentThread.CurrentCulture =
                new System.Globalization.CultureInfo("en-US");  // da-DK            
            Console.WriteLine("Hello, world!");
        }
    }
}

    Tilføj log

    Hvis du har brug for for log kan du tilføje Serilog til projektet ved hjælp af NuGet. Tilføj pakkerne:

    Microsoft.Extensions.Configuration
Microsoft.Extensions.Configuration.Json
Serilog
Serilog.Formatting.Compact
Serilog.Settings.Configuration
Serilog.Sinks.Console
Serilog.Sinks.Debug
Serilog.Sinks.File

    De første pakker gør det muligt at benytte configurationsfiler, og de sidste gør det muligt for Serilog at logge til Console, VS debugger og en fil.

    Tilret nu Program.cs til følgende:

    using Microsoft.Extensions.Configuration;
using Serilog;

namespace MyConsoleApp
{
    internal class Program
    {

        internal static AppSettings? appSettings;
        internal static ILogger? logger;
        // If async use:
        // static async Task Main(string[] args) 
        static void Main()  // If needed add (string[] args)
        {
            // Set culture
            System.Threading.Thread.CurrentThread.CurrentCulture =
                new System.Globalization.CultureInfo("en-US");  // da-DK
            try
            {

                var configuration = new ConfigurationBuilder()
                    .SetBasePath(Directory.GetCurrentDirectory())
                    .AddJsonFile("appsettings.json", true, true)
                    .Build();

                Log.Logger = new LoggerConfiguration()
                    .ReadFrom.Configuration(configuration)
                    .CreateLogger();
                logger = Log.Logger;

                logger.Debug("App start");

                TestClass1 testClass1 = new TestClass1();
                testClass1.Test();
                TestClass2 testClass2 = new TestClass2(logger);
                testClass2.Test();

            }
            catch (Exception ex)
            {
                logger?.Error(ex, "Error in main err handler: " + ex.Message);
            }
            finally
            {
                logger?.Debug("App end");
            }
        }
    }
}

    Koden kræver at Serilog er konfigureret med en appsettings.json-fil. Den skal tilføjes i VS i roden, og når filen er markeret i Solution Explorer skal “Copy to Output directory” i Properties-vinduet sættes til “Copy always”. Filen kan have følgende indhold:

    {
  "Serilog": {
    "Using": [ "Serilog.Sinks.Console", "Serilog.Sinks.File", "Serilog.Sinks.Debug" ],
    "MinimumLevel": "Verbose",
    "WriteTo": [
      {
        "Name": "Console",
        "Args": {
          "outputTemplate": "{Timestamp:HH:mm:ss} {Level:u3} {Message} {NewLine}{Exception}"
        }
      },
      { "Name": "Debug" },
      {
        "Name": "File",
        "Args": { "path": "logs/mylog.log" }
      }
    ]
  }
}

    I filen sættes Serilog til både at skrive til konsol, til en fil og til debuggeren.

    Brug af logger i andre klasser

    Læg mærke til TestClass1 og TestClass2 som på forskelligvis benytter log - den ene gemmer den statiske logger i Program, og den anden gennem contructor:

    internal class TestClass1
{
    public void Test() {
        Program.logger?.Debug("In TestClass1/Test");
    }
}

internal class TestClass2
{
    private readonly ILogger logger;

    public void Test()
    {
        logger?.Debug("In TestClass2/Test");
    }
    public TestClass2(ILogger logger)
    {
        this.logger = logger;
    }
}

    AppSettings

    Hvis du ønsker at benytte AppSettings i applikationen kan du tilføje værdier til appsettings.json filen og mappe disse til en fil. Her er en appsettings-fil med værdier (og Serilog-værdier):

    {
  "AppSettings": {
    "MyInt": 123,
    "MyString": "test123",
    "MyBool": true,
    "MyDate": "2022-01-01T00:00"
  },
  "Serilog": {
    "Using": [ "Serilog.Sinks.Console", "Serilog.Sinks.File", "Serilog.Sinks.Debug" ],
    "MinimumLevel": "Verbose",
    "WriteTo": [
      {
        "Name": "Console",
        "Args": {
          "outputTemplate": "{Timestamp:HH:mm:ss} {Level:u3} {Message} {NewLine}{Exception}"
        }
      },
      { "Name": "Debug" },
      {
        "Name": "File",
        "Args": { "path": "logs/mylog.log" }
      }
    ]
  }
}

    Klassen til at mappe værdier kan se således ud:

    internal class AppSettings
{
    public int MyInt { get; set; }
    public string? MyString { get; set; }
    public bool MyBool { get; set; }
    public DateTime MyDate { get; set; }
}

    Det hele kan bindes sammen i Main (med Serilog):

    using Microsoft.Extensions.Configuration;
using Serilog;

namespace MyConsoleApp
{
    internal class Program
    {

        internal static AppSettings? appSettings;
        internal static ILogger? logger;
        // If async use:
        // static async Task Main(string[] args) 
        static void Main()  // If needed add (string[] args)
        {
            // Set culture
            System.Threading.Thread.CurrentThread.CurrentCulture =
                new System.Globalization.CultureInfo("en-US");  // da-DK
            try
            {

                var configuration = new ConfigurationBuilder()
                    .SetBasePath(Directory.GetCurrentDirectory())
                    .AddJsonFile("appsettings.json", true, true)
                    .Build();

                Log.Logger = new LoggerConfiguration()
                    .ReadFrom.Configuration(configuration)
                    .CreateLogger();
                logger = Log.Logger;

                logger.Debug("App start");

                TestClass1 testClass1 = new TestClass1();
                testClass1.Test();
                TestClass2 testClass2 = new TestClass2(logger);
                testClass2.Test();

                // In VS - remember to clean solution when
                // you change appsettings (see Build-menu)
                logger.Debug("Getting configuration");
                appSettings = new AppSettings();
                configuration.Bind("AppSettings",appSettings);
                logger.Debug($"MyInt: {appSettings.MyInt}");
                logger.Debug($"MyString: {appSettings.MyString}");
                logger.Debug($"MyBool: {appSettings.MyBool}");
                logger.Debug($"MyDate: {appSettings.MyDate}");

            }
            catch (Exception ex)
            {
                logger?.Error(ex, "Error in main err handler: " + ex.Message);
            }
            finally
            {
                logger?.Debug("App end");
            }
        }
    }
}
